1、二维数组定义：
    类型说明符  数组名[常量表达式][常量表达式];
如：  int a[2][3];  // 定义 a 为 3 行 4 列的数组
二维数组是按照“Z”型存储

2、先定义在初始化
    int a[3][4];
    a[0][0] = 1;
    a[2][3] = 9;
  定义的同时初始化
    int a[3][4] = {{1, 2, 3, 4},{5, 6, 7, 8},{9, 0, 1, 2}};
    or
    int a[3][4] = {1, 2, 3, 4, 5, 6, 7, 8, 9, 0, 1, 2};
    or  省去行，但列不能省
    int a[][4] = {{1, 2, 3, 4},{5, 6, 7, 8},{9, 0, 1, 2}};
    or
    int a[][4] = {1, 2, 3, 4, 5, 6, 7, 8, 9, 0, 1, 2};
